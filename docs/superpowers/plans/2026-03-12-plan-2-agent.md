mod cpu;

use std::time::Instant;

use serverbee_common::types::{SystemInfo, SystemReport};
use sysinfo::{Networks, System};

pub struct Collector {
    sys: System,
    networks: Networks,
    prev_net_in: u64,
    prev_net_out: u64,
    prev_time: Instant,
    enable_temperature: bool,
}

impl Collector {
    pub fn new(enable_temperature: bool) -> Self {
        let mut sys = System::new_all();
        sys.refresh_all();
        let networks = Networks::new_with_refreshed_list();

        let (net_in, net_out) = network::total_bytes(&networks);

        Self {
            sys,
            networks,
            prev_net_in: net_in,
            prev_net_out: net_out,
            prev_time: Instant::now(),
            enable_temperature,
        }
    }

    pub fn collect(&mut self) -> SystemReport {
        self.sys.refresh_all();
        self.networks.refresh(true);

        let elapsed = self.prev_time.elapsed().as_secs_f64().max(1.0);
        let (net_in, net_out) = network::total_bytes(&self.networks);
        let net_in_speed = ((net_in - self.prev_net_in) as f64 / elapsed) as i64;
        let net_out_speed = ((net_out - self.prev_net_out) as f64 / elapsed) as i64;

        self.prev_net_in = net_in;
        self.prev_net_out = net_out;
        self.prev_time = Instant::now();

        let temperature = if self.enable_temperature {
            temperature::get_temperature()
        } else {
            None
        };

        SystemReport {
            cpu: cpu::usage(&self.sys),
            mem_used: memory::mem_used(&self.sys),
            swap_used: memory::swap_used(&self.sys),
            disk_used: disk::used(),
            net_in_speed,
            net_out_speed,
            net_in_transfer: net_in as i64,
            net_out_transfer: net_out as i64,
            load1: load::load1(),
            load5: load::load5(),
            load15: load::load15(),
            tcp_conn: process::tcp_connections(),
            udp_conn: process::udp_connections(),
            process_count: process::count(&self.sys),
            uptime: System::uptime(),
            temperature,
            gpu: None,
        }
    }

    pub fn system_info(&self) -> SystemInfo {
        SystemInfo {
            cpu_name: cpu::name(&self.sys),
            cpu_cores: cpu::cores(&self.sys),
            cpu_arch: cpu::arch(),
            os: System::long_os_version().unwrap_or_default(),
            kernel_version: System::kernel_version().unwrap_or_default(),
            mem_total: memory::mem_total(&self.sys),
            swap_total: memory::swap_total(&self.sys),
            disk_total: disk::total(),
            ipv4: None,
            ipv6: None,
            virtualization: None,
            agent_version: serverbee_common::constants::VERSION.to_string(),
        }
    }
}

    {
        count_connections("/proc/net/tcp").unwrap_or(0)
            + count_connections("/proc/net/tcp6").unwrap_or(0)

    #[cfg(not(target_os = "linux"))]

    }

    {
        count_connections("/proc/net/udp").unwrap_or(0)
            + count_connections("/proc/net/udp6").unwrap_or(0)
    }
    #[cfg(not(target_os = "linux"))]
    {
        0
    }
}

#[cfg(target_os = "linux")]
fn count_connections(path: &str) -> std::io::Result<i32> {
    let content = std::fs::read_to_string(path)?;
    // First line is header
    Ok((content.lines().count() - 1).max(0) as i32)

use sysinfo::Components;

pub fn get_temperature() -> Option<f64> {
    let components = Components::new_with_refreshed_list();
    let temps: Vec<f64> = components
        .iter()
        .filter_map(|c| c.temperature().map(|t| t as f64))
        .collect();

    if temps.is_empty() {
        None
    } else {
        Some(temps.iter().sum::<f64>() / temps.len() as f64)
    }
}

use anyhow::Result;
use serde::{Deserialize, Serialize};

use crate::config::AgentConfig;

#[derive(Debug, Serialize)]
struct RegisterRequest {}

#[derive(Debug, Deserialize)]
struct RegisterResponse {
    data: RegisterData,
}

#[derive(Debug, Deserialize)]
struct RegisterData {
    server_id: String,
    token: String,
}

pub async fn register_agent(config: &AgentConfig) -> Result<(String, String)> {
    let url = format!(
        "{}/api/agent/register",
        config.server_url.trim_end_matches('/')
    );

    let client = reqwest::Client::new();
    let resp = client
        .post(&url)
        .bearer_auth(&config.auto_discovery_key)
        .json(&RegisterRequest {})
        .send()
        .await?;

    if !resp.status().is_success() {
        anyhow::bail!("Registration failed: HTTP {}", resp.status());
    }

    let data: RegisterResponse = resp.json().await?;
    Ok((data.data.server_id, data.data.token))
}

pub fn save_token(token: &str) -> Result<()> {
    let path = AgentConfig::config_path();

    let content = if std::path::Path::new(path).exists() {
        std::fs::read_to_string(path)?
    } else {
        String::new()
    };

    // Simple approach: append or replace token line
    let mut lines: Vec<String> = content.lines().map(String::from).collect();
    let token_line = format!("token = \"{}\"", token);

    if let Some(pos) = lines.iter().position(|l| l.starts_with("token")) {
        lines[pos] = token_line;
    } else {
        lines.push(token_line);
    }

    std::fs::write(path, lines.join("\n"))?;
    Ok(())
}

```rust
use std::time::Duration;

use anyhow::Result;
use futures_util::{SinkExt, StreamExt};
use tokio::time;
use tokio_tungstenite::{connect_async, tungstenite::Message};

use serverbee_common::protocol::{AgentMessage, ServerMessage};

use crate::collector::Collector;
use crate::config::AgentConfig;

pub struct Reporter {
    config: AgentConfig,
    collector: Collector,
    reconnect_delay: Duration,
}

impl Reporter {
    pub fn new(config: AgentConfig) -> Self {
        let collector = Collector::new(config.collector.enable_temperature);
        Self {
            config,
            collector,
            reconnect_delay: Duration::from_secs(1),
        }
    }

    pub async fn run(&mut self) {
        loop {
            match self.connect_and_run().await {
                Ok(()) => {
                    tracing::info!("Connection closed normally");
                }
                Err(e) => {
                    tracing::warn!("Connection error: {e}");
                }
            }

            self.backoff_sleep().await;
        }
    }

    async fn connect_and_run(&mut self) -> Result<()> {
        let ws_url = format!(
            "{}/api/agent/ws?token={}",
            self.config
                .server_url
                .replace("http://", "ws://")
                .replace("https://", "wss://")
                .trim_end_matches('/'),
            self.config.token
        );

        tracing::info!("Connecting to {}", self.config.server_url);

        let (ws_stream, _) = connect_async(&ws_url).await?;
        let (mut write, mut read) = ws_stream.split();

        // Wait for Welcome
        let welcome_msg = read
            .next()
            .await
            .ok_or_else(|| anyhow::anyhow!("No welcome message"))??;

        let welcome: ServerMessage = serde_json::from_str(welcome_msg.to_text()?)?;
        let report_interval = match welcome {
            ServerMessage::Welcome {
                report_interval, ..
            } => {
                tracing::info!("Connected, report interval: {}s", report_interval);
                report_interval
            }
            _ => {
                anyhow::bail!("Expected Welcome message, got: {:?}", welcome);
            }
        };

        // Reset reconnect delay on successful connection
        self.reconnect_delay = Duration::from_secs(1);

        // Send SystemInfo
        let sys_info = self.collector.system_info();
        let msg = AgentMessage::SystemInfo {
            msg_id: uuid::Uuid::new_v4().to_string(),
            info: sys_info,
        };
        write
            .send(Message::Text(serde_json::to_string(&msg)?))
            .await?;

        // Main loop
        let mut report_ticker = time::interval(Duration::from_secs(report_interval as u64));

        loop {
            tokio::select! {
                _ = report_ticker.tick() => {
                    let report = self.collector.collect();
                    let msg = AgentMessage::Report(report);
                    let text = serde_json::to_string(&msg)?;
                    write.send(Message::Text(text)).await?;
                }
                msg = read.next() => {
                    match msg {
                        Some(Ok(Message::Text(text))) => {
                            let server_msg: ServerMessage = serde_json::from_str(&text)?;
                            match server_msg {
                                ServerMessage::Ping => {
                                    let pong = serde_json::to_string(&AgentMessage::Pong)?;
                                    write.send(Message::Text(pong)).await?;
                                }
                                ServerMessage::Exec { task_id, command, timeout } => {
                                    tracing::info!("Received exec: {task_id}");
                                    let result = execute_command(&command, timeout).await;
                                    let msg = AgentMessage::TaskResult {
                                        msg_id: uuid::Uuid::new_v4().to_string(),
                                        result,
                                    };
                                    write.send(Message::Text(serde_json::to_string(&msg)?)).await?;
                                }
                                ServerMessage::Ack { msg_id } => {
                                    tracing::debug!("Ack received: {msg_id}");
                                }
                                _ => {
                                    tracing::debug!("Unhandled message: {:?}", server_msg);
                                }
                            }
                        }
                        Some(Ok(Message::Close(_))) => {
                            tracing::info!("Server closed connection");
                            return Ok(());
                        }
                        Some(Err(e)) => {
                            return Err(e.into());
                        }
                        None => {
                            return Ok(());
                        }
                        _ => {}
                    }
                }
            }
        }
    }

    async fn backoff_sleep(&mut self) {
        let jitter_factor = 0.8 + rand::random::<f64>() * 0.4;
        let delay = self.reconnect_delay.mul_f64(jitter_factor);
        tracing::info!("Reconnecting in {:.1}s", delay.as_secs_f64());
        time::sleep(delay).await;

        self.reconnect_delay = std::cmp::min(
            self.reconnect_delay * 2,
            Duration::from_secs(30),
        );
    }
}

async fn execute_command(
    command: &str,
    timeout: Option<u32>,
) -> serverbee_common::types::TaskResult {
    use tokio::process::Command;

    let timeout_secs = timeout.unwrap_or(serverbee_common::constants::DEFAULT_COMMAND_TIMEOUT_SECS);

    let result = time::timeout(
        Duration::from_secs(timeout_secs as u64),
        Command::new("sh").arg("-c").arg(command).output(),
    )
    .await;

    match result {
        Ok(Ok(output)) => {
            let stdout = String::from_utf8_lossy(&output.stdout);
            let stderr = String::from_utf8_lossy(&output.stderr);
            let mut combined = stdout.to_string();
            if !stderr.is_empty() {
                combined.push_str(&stderr);
            }

            // Truncate to max size
            if combined.len() > serverbee_common::constants::MAX_TASK_OUTPUT_SIZE {
                combined.truncate(serverbee_common::constants::MAX_TASK_OUTPUT_SIZE);
                combined.push_str("\n... [truncated]");
            }

            serverbee_common::types::TaskResult {
                task_id: String::new(), // Will be set by caller
                output: combined,
                exit_code: output.status.code().unwrap_or(-1),
            }
        }
        Ok(Err(e)) => serverbee_common::types::TaskResult {
            task_id: String::new(),
            output: format!("Failed to execute: {e}"),
            exit_code: -1,
        },
        Err(_) => serverbee_common::types::TaskResult {
            task_id: String::new(),
            output: format!("Command timed out after {timeout_secs}s"),
            exit_code: -1,
        },
    }
}

mod collector;
mod config;
mod register;
mod reporter;

use tracing_subscriber::EnvFilter;

use crate::config::AgentConfig;
use crate::reporter::Reporter;

#[tokio::main]
async fn main() -> anyhow::Result<()> {
    let mut config = AgentConfig::load().unwrap_or_else(|e| {
        eprintln!("Failed to load config: {e}");
        eprintln!("Please create agent.toml or /etc/serverbee/agent.toml");
        std::process::exit(1);
    });

    tracing_subscriber::fmt()
        .with_env_filter(
            EnvFilter::try_from_default_env()
                .unwrap_or_else(|_| config.log.level.parse().unwrap_or_else(|_| "info".into())),
        )
        .init();

    tracing::info!(
        "ServerBee Agent v{} starting...",
        serverbee_common::constants::VERSION
    );

    // Register if no token
    if config.token.is_empty() {
        if config.auto_discovery_key.is_empty() {
            anyhow::bail!(
                "No token configured and no auto_discovery_key set. \
                 Please set token in config or provide auto_discovery_key for auto-registration."
            );
        }

        tracing::info!("No token found, registering with server...");
        let (_server_id, token) = register::register_agent(&config).await?;
        tracing::info!("Registration successful");

        // Save token to config file
        if let Err(e) = register::save_token(&token) {
            tracing::warn!("Failed to save token to config: {e}");
            tracing::info!("Token: {token}");
        }

        config.token = token;
    }

    // Start reporter
    let mut reporter = Reporter::new(config);
    reporter.run().await;

    Ok(())
}
```
